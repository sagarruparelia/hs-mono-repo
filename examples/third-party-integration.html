<!-- Declarative approach using data attributes -->
<div data-mfe="profile" data-theme="light"></div>
<div data-mfe="summary" data-theme="dark"></div>

<!-- Load the remote entry scripts -->
<script type="module" src="http://localhost:4203/remoteEntry.js"></script>
<script type="module" src="http://localhost:4204/remoteEntry.js"></script>

// Or programmatic approach:
import('http://localhost:4203/bootstrap.js').then(({ mount }) => {
const instance = mount(userDocument.getElementById('profile-container'), {
theme: 'light',
userId: 'user123',
onUpdate: (data) => console.log('Profile updated:', data)
});
});
